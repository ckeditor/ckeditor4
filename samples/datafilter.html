This editor is using a custom configuration for allowed contents. 

CKEDITOR.replace( 'editor2b', {
	allowedContent: {
		'$1': 'strong,em,ul,ol',
		'$2': {
			elements: 'h1,h2,h3,p,blockquote,li',
			styles: 'text-align'
		},
		a: { attrs: 'href,target' },
		img: {
			attrs: {
				src: true,
				alt: true
			},
			classes: 'left,right'
		},
		span: function( element ) {
			// Only span having one of these styles is allowed.
			return element.styles[ 'font-family' ] || element.styles.color;
		}
	}
} );
